    ```typescript
    describe('seat randomization fairness', () => {
      it('seat frequencies are uniform across 100 rounds with 8 players', () => {
        const players = makePlayers(8)  // 2 pods of 4
        const rounds = 100
        // Map<playerId, [seat1Count, seat2Count, seat3Count, seat4Count]>
        const seatCounts = new Map<string, number[]>()
        for (const p of players) {
          seatCounts.set(p.id, [0, 0, 0, 0])
        }

        const previousRounds: RoundHistory[] = []
        for (let r = 0; r < rounds; r++) {
          const result = generatePods(players, previousRounds)
          for (const pod of result.assignments) {
            if (pod.is_bye) continue
            for (const player of pod.players) {
              const counts = seatCounts.get(player.player_id)!
              counts[player.seat_number! - 1]++
            }
          }
          previousRounds.push({
            pods: result.assignments.map(a => ({
              playerIds: a.players.map(p => p.player_id),
              isBye: a.is_bye,
            })),
          })
        }

        // Each player played 100 rounds, expected 25 times per seat
        const expectedPerSeat = rounds / 4
        for (const [playerId, counts] of seatCounts) {
          for (let seatIdx = 0; seatIdx < 4; seatIdx++) {
            const deviation = Math.abs(counts[seatIdx] - expectedPerSeat) / expectedPerSeat
            expect(
              deviation,
              `Player ${playerId} seat ${seatIdx + 1}: got ${counts[seatIdx]}, expected ~${expectedPerSeat}`
            ).toBeLessThan(0.15)
          }
        }
      })

      it('seat frequencies are uniform across 100 rounds with 12 players', () => {
        const players = makePlayers(12)  // 3 pods of 4
        const rounds = 100
        // Map<playerId, [seat1Count, seat2Count, seat3Count, seat4Count]>
        const seatCounts = new Map<string, number[]>()
        for (const p of players) {
          seatCounts.set(p.id, [0, 0, 0, 0])
        }

        const previousRounds: RoundHistory[] = []
        for (let r = 0; r < rounds; r++) {
          const result = generatePods(players, previousRounds)
          for (const pod of result.assignments) {
            if (pod.is_bye) continue
            for (const player of pod.players) {
              const counts = seatCounts.get(player.player_id)!
              counts[player.seat_number! - 1]++
            }
          }
          previousRounds.push({
            pods: result.assignments.map(a => ({
              playerIds: a.players.map(p => p.player_id),
              isBye: a.is_bye,
            })),
          })
        }

        // Each player played 100 rounds, expected 25 times per seat
        const expectedPerSeat = rounds / 4
        for (const [playerId, counts] of seatCounts) {
          for (let seatIdx = 0; seatIdx < 4; seatIdx++) {
            const deviation = Math.abs(counts[seatIdx] - expectedPerSeat) / expectedPerSeat
            expect(
              deviation,
              `Player ${playerId} seat ${seatIdx + 1}: got ${counts[seatIdx]}, expected ~${expectedPerSeat}`
            ).toBeLessThan(0.15)
          }
        }
      })
    })
    ```

    **Step 2: Analyze surviving mutants**

    Review the Stryker output. For each surviving mutant:
    - Identify the mutated line and the mutation type (arithmetic, conditional, string, etc.)
    - Determine if the mutation is in critical path code (quadratic scoring, multi-start loop, swap pass comparison, bye selection)
    - Critical path mutants MUST be killed. Non-critical (e.g., log messages, comments) can survive.

    **Step 3: Strengthen tests to kill surviving mutants**

    Common surviving mutants and how to kill them:

    a. **Arithmetic mutations in quadratic scoring** (`encounters * encounters` -> `encounters + encounters` or `encounters * 0`):
       - Add test: `getOpponentScore` with encounters=2 returns 4 (not 4 from `2+2`, but specifically test encounters=3 returns 9 vs `3+3=6`)
       - Ensure test values distinguish `*` from `+` (encounters >= 3 is the key case)

    b. **Conditional boundary mutations in swap pass** (`<` -> `<=` or `<` -> `===`):
       - Add test where swapped score equals current score -- verify NO swap occurs (strict less-than)
       - Add test where swapped score is less by exactly 1 -- verify swap DOES occur

    c. **Loop boundary mutations in multi-start** (`i < NUM_STARTS` -> `i <= NUM_STARTS` or `i < 0`):
       - Test that multi-start runs exactly NUM_STARTS iterations (mock shuffleArray to count calls)

    d. **Removal mutations** (deleting `improved = true` in swap pass):
       - Test a scenario where the first swap enables a second swap -- verify both swaps happen

    e. **Return value mutations** (returning empty array instead of computed result):
       - Every function must have at least one test asserting non-empty return value with meaningful content

    **Step 4: Re-run Stryker to confirm >=80% score**

    ```bash
    npx stryker run --mutate "src/lib/pod-algorithm.ts"
    ```
